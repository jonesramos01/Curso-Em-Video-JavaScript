MÓDULO F: VARIÁVEIS COMPOSTAS = ARRAYS = VETORES

        var a = [a0, a1, a2, ... ]
        
        a = Array/Vetor
        0, 1, 2, ... = Índice/Chave
        [a0, a1, a2, ... ] = Elemento
        a0 = Conteúdo/Valor 

        num[3] = 7 //Adiciona o valor 7 a posição 3 no vetor. 
        num.push(7) //Adiciona o valor 7 a última posição, seja ela qual for.
        num.length //Para saber o comprimento do Array. "Quantidade de elementos/posições". Obs: No JS é sem ()
        num.sort() //Ordena os elementos em ordem crescente.
